---
title: "Get Sweep History"
description: "Retrieve janitor sweep history"
api: "GET /api/v1/workers/janitor/history"
---

Returns a chronological history of janitor sweeps, including both automatic daemon sweeps and manual API-triggered operations.

<ParamField query="limit" type="number" default={10}>
  Maximum number of sweep records to return. Maximum allowed: 100.
</ParamField>

<RequestExample>

```bash Default Limit
curl -X GET https://alfred-acme.tailnet.ts.net:3100/api/v1/workers/janitor/history \
  -H "Authorization: Bearer YOUR_API_KEY"
```

```bash Custom Limit
curl -X GET "https://alfred-acme.tailnet.ts.net:3100/api/v1/workers/janitor/history?limit=25" \
  -H "Authorization: Bearer YOUR_API_KEY"
```

</RequestExample>

<ResponseField name="status" type="string">
  Response status: `ok` or `error`
</ResponseField>

<ResponseField name="total_sweeps" type="number">
  Total number of sweeps in history
</ResponseField>

<ResponseField name="returned" type="number">
  Number of sweep records returned in this response
</ResponseField>

<ResponseField name="sweeps" type="object[]">
  List of sweep records, most recent first

  <ResponseField name="sweep_id" type="string">
    Unique identifier for the sweep
  </ResponseField>

  <ResponseField name="timestamp" type="string">
    ISO 8601 timestamp when sweep started
  </ResponseField>

  <ResponseField name="type" type="string">
    Sweep type: `auto` (daemon), `manual` (API), or `cli`
  </ResponseField>

  <ResponseField name="operation" type="string">
    Operation performed: `scan` or `fix`
  </ResponseField>

  <ResponseField name="duration_seconds" type="number">
    Time taken to complete the sweep
  </ResponseField>

  <ResponseField name="files_scanned" type="number">
    Number of vault files scanned
  </ResponseField>

  <ResponseField name="issues_found" type="number">
    Number of issues detected
  </ResponseField>

  <ResponseField name="issues_fixed" type="number">
    Number of issues fixed (0 for scan-only operations)
  </ResponseField>

  <ResponseField name="triggered_by" type="string" optional>
    Trigger source for manual sweeps: `api`, `cli`, or username
  </ResponseField>
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp of history retrieval
</ResponseField>
