---
title: "Get Distiller Status"
description: "Retrieve distiller extraction status and statistics"
api: "GET /api/v1/workers/distiller/status"
---

Returns the current status of the distiller daemon, including the most recent extraction run, candidate statistics, and extraction metrics.

<RequestExample>

```bash cURL
curl -X GET https://alfred-acme.tailnet.ts.net:3100/api/v1/workers/distiller/status \
  -H "Authorization: Bearer YOUR_API_KEY"
```

</RequestExample>

<ResponseExample>

```json Response
{
  "status": "ok",
  "daemon_running": true,
  "last_run": {
    "timestamp": "2026-02-26T09:00:00Z",
    "duration_seconds": 142.5,
    "records_scanned": 423,
    "candidates_found": 8,
    "learnings_extracted": 12,
    "run_type": "auto"
  },
  "extraction_stats": {
    "total_learnings": 87,
    "by_type": {
      "assumption": 23,
      "decision": 31,
      "constraint": 18,
      "contradiction": 9,
      "synthesis": 6
    }
  },
  "current_candidates": {
    "projects": 3,
    "tasks": 2,
    "issues": 2,
    "notes": 1,
    "total": 8
  },
  "next_run_in_seconds": 7200,
  "timestamp": "2026-02-26T10:30:00Z"
}
```

</ResponseExample>

<ResponseField name="status" type="string">
  Overall status: `ok` or `error`
</ResponseField>

<ResponseField name="daemon_running" type="boolean">
  Whether the distiller daemon is currently running
</ResponseField>

<ResponseField name="last_run" type="object">
  Information about the most recent extraction run

  <ResponseField name="timestamp" type="string">
    ISO 8601 timestamp when run started
  </ResponseField>

  <ResponseField name="duration_seconds" type="number">
    How long the extraction took
  </ResponseField>

  <ResponseField name="records_scanned" type="number">
    Number of vault records scanned for candidates
  </ResponseField>

  <ResponseField name="candidates_found" type="number">
    Number of records identified as extraction candidates
  </ResponseField>

  <ResponseField name="learnings_extracted" type="number">
    Number of learning records created
  </ResponseField>

  <ResponseField name="run_type" type="string">
    Type of run: `auto` (daemon), `manual` (API), or `cli`
  </ResponseField>
</ResponseField>

<ResponseField name="extraction_stats" type="object">
  Cumulative extraction statistics

  <ResponseField name="total_learnings" type="number">
    Total number of learning records in vault
  </ResponseField>

  <ResponseField name="by_type" type="object">
    Breakdown by learning type
  </ResponseField>
</ResponseField>

<ResponseField name="current_candidates" type="object">
  Breakdown of current extraction candidates by source type
</ResponseField>

<ResponseField name="next_run_in_seconds" type="number">
  Seconds until the next automatic extraction (only present if daemon is running)
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp of status check
</ResponseField>
