---
title: "Extract Learnings"
description: "Scan vault and extract learnings using the agent"
api: "POST /api/v1/workers/distiller/run"
---

Performs a scan for extraction candidates followed by agent-based learning extraction. The distiller agent will analyze candidate records and create new learning records (assumptions, decisions, constraints, contradictions, synthesis) for each piece of latent knowledge discovered.

<ParamField body="project" type="string" optional>
  Filter extraction to a specific project. If omitted, extracts from all candidates.

  Example: `"acme-redesign"`
</ParamField>

<RequestExample>

```bash Extract All Candidates
curl -X POST https://alfred-acme.tailnet.ts.net:3100/api/v1/workers/distiller/run \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
```

```bash Extract from Specific Project
curl -X POST https://alfred-acme.tailnet.ts.net:3100/api/v1/workers/distiller/run \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "project": "acme-redesign"
  }'
```

</RequestExample>

</ResponseField>

<ResponseField name="scan_id" type="string">
  Unique identifier for the scan phase
</ResponseField>

<ResponseField name="run_id" type="string">
  Unique identifier for the extraction phase
</ResponseField>

<ResponseField name="scan_duration_seconds" type="number">
  Time taken for the scan phase
</ResponseField>

<ResponseField name="extraction_duration_seconds" type="number">
  Time taken for the extraction phase
</ResponseField>

<ResponseField name="total_duration_seconds" type="number">
  Total operation time
</ResponseField>

<ResponseField name="records_scanned" type="number">
  Number of vault records scanned
</ResponseField>

<ResponseField name="candidates_found" type="number">
  Number of extraction candidates identified
</ResponseField>

<ResponseField name="candidates_processed" type="number">
  Number of candidates actually processed by agent
</ResponseField>

<ResponseField name="learnings_extracted" type="number">
  Total number of learning records created
</ResponseField>

<ResponseField name="learnings_by_type" type="object">
  Breakdown of extracted learnings by type
</ResponseField>

<ResponseField name="extractions" type="object[]">
  Detailed list of extractions per source record

  <ResponseField name="source_file" type="string">
    Vault-relative path to the source record
  </ResponseField>

  <ResponseField name="source_type" type="string">
    Source record type
  </ResponseField>

  <ResponseField name="learnings_created" type="number">
    Number of learnings extracted from this source
  </ResponseField>

  <ResponseField name="learnings" type="object[]">
    List of learnings created

    <ResponseField name="type" type="string">
      Learning type: `assumption`, `decision`, `constraint`, `contradiction`, `synthesis`
    </ResponseField>

    <ResponseField name="title" type="string">
      Learning title
    </ResponseField>

    <ResponseField name="file" type="string">
      Vault-relative path to created learning record
    </ResponseField>

    <ResponseField name="rationale" type="string">
      Brief explanation of the learning
    </ResponseField>
  </ResponseField>
</ResponseField>

<ResponseField name="deduplication" type="object">
  Deduplication statistics

  <ResponseField name="checked" type="number">
    Number of learnings checked for duplicates
  </ResponseField>

  <ResponseField name="duplicates_found" type="number">
    Number of duplicates detected
  </ResponseField>

  <ResponseField name="duplicates_merged" type="number">
    Number of duplicates merged into existing records
  </ResponseField>
</ResponseField>

<ResponseField name="project_filter" type="string | null">
  Project filter applied (null if extracting from all candidates)
</ResponseField>

<ResponseField name="mutations_logged" type="number">
  Number of vault mutations written to audit log
</ResponseField>

<ResponseField name="timestamp" type="string">
  ISO 8601 timestamp of operation completion
</ResponseField>
